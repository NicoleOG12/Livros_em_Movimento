create database Gerenciamento;
use Gerenciamento;

create table usuário (
ID_usuário integer auto_increment unique,
nome_usuário varchar(50),
email_usuário varchar(50),
senha varchar(50),
gêneros_de_interesse varchar (100),
primary key (ID_usuário)
);

create table livros (
ID_livros integer auto_increment unique,
nome_livro varchar(50),
autor_livro varchar(50),
gênero int,
ID_usuário int,
disponibilidade boolean,
primary key (ID_livros),
foreign key (gênero) references categorias (ID_categoria)
);
drop table livros;

create table categorias (
ID_categoria integer unique auto_increment,
gênero varchar(50)
);

create table empréstimo (
ID_empréstimo integer auto_increment unique,
ID_livro int ,
ID_solicitação int ,
data_de_solicitação date,
ID_usuário INT ,
status_da_solicitação enum('pendente', 'aceito', 'recusado'),
data_de_devolução date, 
primary key (ID_empréstimo)
);

create table avaliações (
ID_avaliações integer auto_increment unique,
ID_usuário_avaliador int,
ID_usuário_remetente int,
comentario_do_avaliador varchar (100),
classificação int,
primary key (ID_avaliações),
foreign key (ID_usuário_avaliador) references usuário (ID_usuário),
foreign key (ID_usuário_remetente) references usuário (ID_usuário)
);
